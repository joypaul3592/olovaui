## Usage

```tsx
import { Avatar, AvatarFallback, AvatarImage } from "@/app/components/avatar/avatar";

export default function Example() {
  return (
    <Avatar>
      <AvatarImage src="https://olova.net/img/olova.png" alt="Olova UI" />
      <AvatarFallback>OU</AvatarFallback>
    </Avatar>
  );
}
```
