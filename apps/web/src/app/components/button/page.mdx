## Usage

```tsx
import { ArrowUpRight } from "lucide-react";
import { Button } from "@/app/components/button/button";

export default function Example() {
  return (
    <Button variant="outline" rightIcon={<ArrowUpRight className="h-4 w-4" />}>
      Button
    </Button>
  );
}
```
